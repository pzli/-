2、4归并排序
class MergeSort {
public:
    int* mergeSort(int* A, int n) {
        if (n >= 2) {
            mergeRecursion(A, 0, n - 1);
        }
        return A;
    }
    void mergeRecursion(int* A, int start, int end) {
        if (start >= end) {
            return;
        }
        int mid = (start + end) >> 1;
        mergeRecursion(A, start, mid);
        mergeRecursion(A, mid + 1, end);
        // 先将数组拆分成单个
        mergeOption(A, start, mid, end);
        // 将拆分的 2 个数组按照规则合并成长度为两个数组相加的排好序的数组，直到整个数组有序
    }
    void mergeOption(int* A, int start, int mid, int end) {
        int i = start, j = mid + 1;
        int k = 0;
        int *temp = (int *)malloc((end - start+ 1) * sizeof(int));
        // 新开数组 存排序结果

        while (i <= mid && j <= end) {
            A[i] > A[j] ? temp[k++] = A[j++] : temp[k++] = A[i++];
        } 
        while (i <= mid) {
            temp[k++] = A[i++];
        }
        while (j <= mid) {
            temp[k++] = A[j++];
        }
        // 两个数组依次选两个中最小的放入 temp
        for (i = 0; i < k; i++) {
            A[start + i] = temp[i];
        }
        // 排好的序列放到原数组的对应位置

        free(temp);
    }
};
2、5快速排序
class QuickSort {
public:
    int* quickSort(int* A, int n) {
        // write code here
        if(n >= 2) quick_sort(A, 0, n-1);

        return A;
    }

    void quick_sort(int *A, int low, int high){
        if(low>=high)return;
        int r = partition(A,low,high);
        quick_sort(A, low, r-1);
        quick_sort(A, r+1, high);
    }

    int partition(int *A, int l, int h){

        int key = A[l];
        while(l<h){
            while(l<h && A[h]>=key) h--;
            A[l] = A[h];
            while(l<h && A[l]<=key) l++;
            A[h] = A[l];
        }

        A[l] = key;
        return l;
    }
};

2、8基数排序和基数排序都是基于桶排序的思想

总结：
  • O（N^2）的排序有冒泡排序、插入排序、选择排序
  • O（N*logN）的排序有归并排序、堆排序、快速排序
  • 基于桶排序的排序算法时间复杂度近似于O（N），但是要知道元素的范围区间给定具体数量的桶

  • 对于空间复杂度，冒泡排序、插入麦序、选择排序、非递归实现的堆排序和希尔排序为O（1）
  • 归并排序为O（logN）、递归实现的堆排序空间复杂度也为O（logN）（函数栈）
  • 快速排序根据选择的key是O（N）~O（logN）
  • 基数排序和计数排序空间复杂度为O（M），M为桶的数量


  • 不稳定的排序算法为选择排序、堆排序、快速排序、希尔排序

4、KMP算法
    void getNext(vector<int> &next, string &T)
    {
        int lent = T.size();
        int i = -1, j = 0;
        next[0] = -1;
         
        while(j < lent)
        {
            if(i == -1 || T[i] == T[j])
            {
                ++i;
                ++j;
                next[j] = i;
            }
            else
                i = next[i];
        }
    }
     
    int KMP(string S, string T)
    {
        int i = 0, j = 0;
        int lens = S.size();
        int lent = T.size();
         
        //next数组
        vector<int> next(lent + 1, 0);
        getNext(next, T);
         
        //KMP
        while((i < lens) && (j < lent))
        {
            if((j == -1) || (S[i] == T[j]))
            {
                ++i;
                ++j;
            }
            else
                j = next[j];
        }
         
        if(j == lent)
            return i - lent;
        else
            return -1;
         
    }
